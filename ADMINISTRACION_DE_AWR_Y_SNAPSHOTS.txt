********* GENERAR UN SNAPSHOTS MANUAL *********

1.1- Verificar el snapshots current 

col BEGIN_INTERVAL_TIME format a40
col END_INTERVAL_TIME format a30
select snap_id,BEGIN_INTERVAL_TIME,END_INTERVAL_TIME 
  from dba_hist_snapshot 
 where BEGIN_INTERVAL_TIME > systimestamp -1 
 order by BEGIN_INTERVAL_TIME desc;


1.1-Verificar el snapshots current de varios dias

set lines 100 pages 999
select DBID,snap_id,
  snap_level,
  to_char(begin_interval_time, 'dd/mm/yy hh24:mi:ss') begin
from dba_hist_snapshot
order by 1;


2.- Ejecutar snapshots manual

EXEC DBMS_WORKLOAD_REPOSITORY.create_snapshot;


3.- Workload Repository Reports creados 

col NAME format a20
SELECT name,
  detected_usages,
  currently_used,
  TO_CHAR(last_sample_date,'DD-MON-YYYY:HH24:MI') last_sample
FROM dba_feature_usage_statistics
WHERE name = 'AWR Report';

********* CONFIGURAR EL STATISTICS_LEVEL PARA LA RECOLECCION DE MÉTRICAS *********

Puede ser cualquiera de estos, dependiendo de las necesidades. Se recomienda que esté en ALL para que recolecte métricas de S.O.

1.- Consultar parametro

show parameter statistics_level
show parameter statis

3.-Modificar parametro (recomendación ALL)

ALTER SYSTEM SET statistics_level=basic;
ALTER SYSTEM SET statistics_level=typical;
ALTER SYSTEM SET statistics_level=all;


********** Consultar el intervalo de retención en Minutos **********

select
   extract( day from snap_interval) *24*60+
   extract( hour from snap_interval) *60+
   extract( minute from snap_interval ) "Snapshot Interval",
   extract( day from retention) *24*60+
   extract( hour from retention) *60+
   extract( minute from retention ) "Retention Interval"
from
   dba_hist_wr_control;


********** Consultar el intervalo de retención dias **********

col RETENTION format a30
col SNAP_INTERVAL format a30
select DBID,snap_interval
    ,retention 
  from dba_hist_wr_control;



********** Configurar el intervalo de métricas (minutos) y el tiempo de retención de snapshots (minutos) **********

Default: execute dbms_workload_repository.modify_snapshot_settings (interval => 60,retention => 4320);

#Ejemplo: intervalo para ambientes de laboratorio/pruebas
execute dbms_workload_repository.modify_snapshot_settings (interval => 10,retention => 11520);


#Ejemplo: intervalo de capturas cada 60 minutos y la retención de 525600 minutos o 1 año
execute dbms_workload_repository.modify_snapshot_settings (interval => 60,retention => 525600);


#Ejemplo: intervalo de capturas cada 15 minutos y la retención de 15 dias
execute dbms_workload_repository.modify_snapshot_settings (interval => 15,retention => 21600);


E#jemplo: intervalo de capturas cada 15 minutos y la retención de 30 dias
execute dbms_workload_repository.modify_snapshot_settings (interval => 15,retention => 43200);


IMPORTANTE:
    Cuando el valor de retención a establecer es menor al valor por Default de 8 dias establecido por Oracle, se debe modificar
    el BaseLine Windown Size con el siguiente comando:

--- CONSULTAR EL WINDOW_SIZE
select baseline_type,moving_window_size from dba_hist_baseline;

--- MODIFICAR EL WINDOW_SIZE
execute dbms_workload_repository.modify_baseline_window_size(window_size=> 7);



*************** Borrar snapshots de forma manual****************************************

EXEC dbms_workload_repository.drop_snapshot_range (low_snap_id=>1107, high_snap_id=>1108);
