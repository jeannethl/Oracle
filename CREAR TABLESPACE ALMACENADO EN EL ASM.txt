----------------- CREAR TABLESPACE ALMACENADO EN EL ASM -----------------------

--->ESTE TABLESPACE SE GUARDA EN EL ASM +DATA DE UN TAMAÑO INICIAL DE 15MB Y SE AUTOEXTIENDE A LOS SIGUIENTES 20MB 


CREATE TABLESPACE "RADIX" LOGGING DATAFILE '+DATA' SIZE 15M AUTOEXTEND ON NEXT 20M EXTENT MANAGEMENT LOCAL SEGMENT SPACE MANAGEMENT AUTO;
---- Creacion de tablespace bigfile ASM y Filesystem ----

CREATE BIGFILE TABLESPACE BIGTBS2 DATAFILE '/u01/app/oracle/oradata/OCM/bigtbs02.dbf'
SIZE 100M AUTOEXTEND ON NEXT 100M MAXSIZE 100G;

CREATE BIGFILE TABLESPACE BIGTBS1 DATAFILE '+DATA' SIZE 100M AUTOEXTEND ON NEXT 100M MAXSIZE 100G;
----- Cuando usan encriptacion ---
CREATE SMALLFILE TABLESPACE "RADIX" LOGGING DATAFILE '+DATA' SIZE 15M AUTOEXTEND ON NEXT 20M ENCRYPTION USING 'AES192' EXTENT MANAGEMENT LOCAL SEGMENT SPACE MANAGEMENT AUTO DEFAULT NOCOMPRESS STORAGE(ENCRYPT);
