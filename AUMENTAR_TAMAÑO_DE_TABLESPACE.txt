---Crear un tablespaces con varios datafiles autoextends
CREATE TABLESPACE TBS_DBA 
         DATAFILE '+DATA' size 10M,
                  '+DATA' size 10M,
                  '+DATA' size 10M,
                  '+DATA' size 10M
       AUTOEXTEND ON NEXT 10M MAXSIZE 100M;
       

--- Agregar datafile para aumentar el tamaño de un tablespace, autoextendible con un maximo de 32GB
ALTER TABLESPACE RADIX_EVENTLOG ADD DATAFILE '+DATA' SIZE 10G AUTOEXTEND ON NEXT 1G MAXSIZE 32767M;

--- Agregar mas de un datafile a un tablespaces 
ALTER TABLESPACE TBS_DBA 
    ADD DATAFILE '+DATA' SIZE 10M,
                 '+DATA' SIZE 10M
    AUTOEXTEND ON NEXT 10M MAXSIZE 100M;
