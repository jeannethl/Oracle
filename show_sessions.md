## Ver sesiones de usuarios y tipo de cliente que usa

```sql
SET LINESIZE 180 pages 300
CLEAR COLUMNS
CLEAR BREAKS
COL SPID FORMAT A10
COL USERNAME FORMAT A10
COL PROGRAM FORMAT A30
COL "LOGON_TIME" FORMAT A20
COL "DB_USER" FORMAT A20
COL "SO_USER" FORMAT A20
COL MACHINE FORMAT A20
SELECT s.SID
, s.SERIAL#
, p.SPID
, p.PID
, s.USERNAME AS "DB_USER"
, s.OSUSER AS "SO_USER"
, s.MACHINE
, s.PROGRAM
, TO_CHAR(s.LOGON_TIME,'DD-MM-RRRR HH24:MI:SS') AS "LOGON_TIME"
FROM GV$SESSION s
JOIN GV$PROCESS p ON p.ADDR = s.PADDR
AND p.INST_ID = S.INST_ID
WHERE s.TYPE != 'BACKGROUND'
AND s.USERNAME NOT IN ('SYS', 'SYSRAC')
-- AND s.USERNAME = 'PROTOCOLOVB'
--and s.STATUS!='INACTIVE'
--AND s.PROGRAM like '%ISTBDNAIGUATACA%'
order by "LOGON_TIME"
/
```

## Sesiones por nodo y LOGON_TIME

```sql
CLEAR COLUMNS
CLEAR BREAKS
SET LINESIZE 200
COL USERNAME FORMAT A12
COL PROGRAM FORMAT A25
COL "LOGON_TIME" FORMAT A20
COL "DB_USER" FORMAT A20
COL "SO_USER" FORMAT A18
COL MACHINE FORMAT A25
COL CANT_SESSIONS FORMAT 999,999,999
--BREAK ON REPORT ON MACHINE SKIP 2
BREAK ON REPORT ON MACHINE SKIP PAGE
COMPUTE SUM LABEL TOTAL_SESSIONS OF CANT_SESSIONS ON REPORT
COMPUTE SUM LABEL SESSION_FOR_NODO OF CANT_SESSIONS ON MACHINE
select count (*) CANT_SESSIONS
, SUBSTR(s.MACHINE,1,25) MACHINE
, s.USERNAME
, s.OSUSER AS "SO_USER"
, s.PROGRAM
, TO_CHAR(s.LOGON_TIME,'DD-MM-RRRR HH24:MI') AS "LOGON_TIME"
FROM GV$SESSION s
JOIN GV$PROCESS p ON p.ADDR = s.PADDR
AND p.INST_ID = S.INST_ID
WHERE s.TYPE != 'BACKGROUND'
AND S.USERNAME !='SYS'
-- and s.STATUS='INACTIVE'
--AND s.USERNAME ='USRPORTAL'
group by s.MACHINE, s.USERNAME, s.PROGRAM, s.OSUSER, TO_CHAR(s.LOGON_TIME,'DD-MM-RRRR HH24:MI')
order by "LOGON_TIME", MACHINE
/
```


## Cantidad de sesiones por nodo

```sql
CLEAR COLUMNS
CLEAR BREAKS
CLEAR COMPUTES
set linesize 180, pages 500
BREAK ON MACHINE SKIP PAGE 2
BREAK ON REPORT ON MACHINE SKIP PAGE
COLUMN MACHINE NEW_VALUE MACHINE_VAR NOPRINT
TTITLE LEFT "NODO: " MACHINE_VAR SKIP 2
COL USERNAME FORMAT A12
COL PROGRAM FORMAT A25
COL "DB_USER" FORMAT A20
COL "SO_USER" FORMAT A18
COL MACHINE FORMAT A25
COL SESSIONS FORMAT 999,999,999
COL MACHINE NEW_VALUE MACHINE_VAR NOPRINT
COMPUTE SUM LABEL "GRAND TOTAL: " OF SESSIONS ON REPORT
COMPUTE SUM LABEL SESSION_FOR_NODO OF SESSIONS ON MACHINE
select count (*) SESSIONS
, TO_CHAR(s.LOGON_TIME,'DD-MM-RRRR HH24:MI') AS "LOGON_TIME"
, SUBSTR(s.MACHINE,1,31) MACHINE
, s.USERNAME "DB_USER"
, s.OSUSER AS "SO_USER"
, s.PROGRAM
FROM GV$SESSION s
JOIN GV$PROCESS p ON p.ADDR = s.PADDR
AND p.INST_ID = S.INST_ID
WHERE s.TYPE != 'BACKGROUND'
--and s.STATUS='INACTIVE'
AND s.USERNAME not in ('SYS','SYSRAC','SYSTEM')
group by s.MACHINE, s.USERNAME, s.PROGRAM, s.OSUSER, TO_CHAR(s.LOGON_TIME,'DD-MM-RRRR HH24:MI')
order by "LOGON_TIME"
/
```

## Session desde hace un mes

```sql
set linesize 190
set pages 200
col METRIC_NAME for a40
col BEGIN_TIME for a20
col END_TIME for a20
COL METRIC_UNIT for a20
select BEGIN_TIME
, END_TIME
, MAX(MAXVAL) MAXVAL
, MIN(MAXVAL) MINVAL
--, ROUND(AVERAGE,2) AVERAGE
from dba_hist_sysmetric_summary 
where  METRIC_ID=2143 
and INSTANCE_NUMBER=1 
group by BEGIN_TIME, END_TIME, MAXVAL, MINVAL, AVERAGE
order by BEGIN_TIME;
```


```sql
select MAX(MAXVAL) MAXVAL
, MIN(MAXVAL) MINVAL
, to_char(BEGIN_TIME,'DD-MM-RRRR HH24:MI') BEGIN_TIME
from dba_hist_sysmetric_summary 
where  METRIC_ID=2143 
and INSTANCE_NUMBER=1 
and to_char(BEGIN_TIME,'DD-MM-RRRR HH24:MI') > to_char(sysdate-1,'DD-MM-RRRR HH24:MI')
group by to_char(BEGIN_TIME,'DD-MM-RRRR HH24:MI')
order by BEGIN_TIME;
```

## Otro query 

```sql
SELECT C.SQL_TEXT,
B.NAME,
COUNT(1),
SUM(TIME_WAITED)
FROM v$ACTIVE_SESSION_HISTORY A,
v$EVENT_NAME B,
v$SQLAREA C
WHERE to_char(A.SAMPLE_TIME,'DD-MM-RRRR HH24:MI:SS') 
BETWEEN '30-05-2019 10:57:00' AND '30-05-2019 11:58:00' 
AND A.EVENT# = B.EVENT# 
AND A.SESSION_ID= 123 
AND A.SQL_ID = C.SQL_ID
GROUP BY C.SQL_TEXT, B.NAME
/
```

## Sesiones por application

```sql
set line 180
col "Nodo" format a16
col program format a35
col service_name format a16
col CANT format 99999
select (SELECT I.INSTANCE_NAME FROM GV$INSTANCE I WHERE I.INST_ID=b.INST_ID) "Nodo"
, b.program
, b.service_name
, COUNT(1) CANT
from gv$session b
where b.type != 'BACKGROUND'
and UPPER(b.program) not in
('SQL DEVELOPER'
	,'PLSQLDEV.EXE'
	,'TOAD.EXE'
	,'SQLNAVIGATOR.EXE'
	,'SQLNAV5.EXE'
	,'SQLPLUS.EXE'
	,'SQLPLUSW.EXE'
	,'PLSQLDEV 5.0.EXE'
	,'DBART.EXE')
group by b.inst_id, b.program, b.service_name
order by  b.program, b.inst_id, b.service_name; 
```
