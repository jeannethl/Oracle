      T.SQL_ID
     ,S.SNAP_ID
     ,S.INSTANCE_NUMBER
     ,S.MODULE
     ,S.ACTION  
     ,(SN.BEGIN_INTERVAL_TIME) BEGIN_INTERVAL_TIME
     ,(SN.END_INTERVAL_TIME) END_INTERVAL_TIME
--     ,(CAST (DBMS_LOB.SUBSTR (T.SQL_TEXT, 2000) AS NVARCHAR2 (2000)))   SQL_TEXT
     ,(S.EXECUTIONS_DELTA)                                              EXECUTIONS
     ,(S.ELAPSED_TIME_DELTA)                                            ELAPSED_TIME
     ,s.rows_processed_total
     ,s.rows_processed_delta
     ,P.MACHINE
   --,dhsp.TIMESTAMP
FROM DBA_HIST_SQLSTAT S
    ,DBA_HIST_SQLTEXT T
    ,DBA_HIST_SNAPSHOT SN
    ,DBA_HIST_ACTIVE_SESS_HISTORY P
    --,dba_hist_sql_plan dhsp
WHERE T.SQL_ID = S.SQL_ID
 AND S.SNAP_ID = SN.SNAP_ID
 AND S.SNAP_ID = P.SNAP_ID
 AND S.INSTANCE_NUMBER=SN.INSTANCE_NUMBER
 AND S.DBID=SN.DBID
 AND t.sql_id = '0xc5cx75fmgcs'
